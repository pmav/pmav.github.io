<?php

 */

 class SimpleAuth {
 
  private $password;


  function __construct($password) {
   $this->password = $password;
  }


  function isValid($key = 'key') {
   return ($_GET[$key] === md5($this->password) || $_GET[$key] === $this->password);
  }


  function getSecret() {
   return md5($this->password);
  }

 }

?>
